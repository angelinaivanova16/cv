function guessNumber() {
  let rightAnswer = Math.floor(Math.random() * 10) + 1;
  let attempts = 5;

  function writeAnswer(){
    let userAnswer = +prompt('Угадай число от 1 до 10');
    attempts = attempts - 1;

    if (userAnswer == rightAnswer) {
    let again = confirm ('Вы угадали!!! Хотели бы сыграть еще?');
    (again === true) ? guessNumber() : alert ('Игра окончена.');
    } else {
    if (userAnswer > rightAnswer && attempts > 0) {
        alert (`Загаданное число меньше, осталось попыток: ${attempts}`);
    } else if(userAnswer < rightAnswer && userAnswer > 0 && attempts > 0) {
        alert (`Загаданное число больше, осталось попыток: ${attempts}`)
    } else if (isNaN(userAnswer) && attempts > 0) {
        alert ('Введи число!')
    } else if (userAnswer === 0) {
        alert ('Игра окончена.');
        return;
    } else if (attempts == 0) {
        let again2 = confirm ('Попытки закончились, сыграем еще?');
        (again2 === true) ? guessNumber() : alert ('Игра окончена.');
        return;
    }
        return writeAnswer();
    }
  }

  writeAnswer();

  };
